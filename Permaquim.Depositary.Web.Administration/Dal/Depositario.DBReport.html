-- ======================================================================
-- Author:		Germ√°n Sivit

-- Description:	Devuelve un resultset con todos los datos del depositario
-- Uso: EXEC [Dispositivo].[ObtenerDepositarios]
-- ======================================================================
CREATE PROCEDURE [Dispositivo].[ObtenerDepositarios]

AS
BEGIN

DECLARE @SemaforoOnline BIT= CAST(1 AS BIT)
DECLARE @SemaforoAnomalias BIT= CAST(0 AS BIT)
    
	SELECT
		D.Id AS DepositarioId,
		D.Nombre,
		D.Descripcion,
		SEC.Descripcion AS Sector,
		SUC.Descripcion AS Sucursal,
		E.Descripcion AS Empresa,
		D.NumeroSerie,
		D.CodigoExterno,
		M.Nombre AS Modelo,
		D.Habilitado,
		UC.Nombre + ' ' + UC.Apellido AS UsuarioCreacion,
		D.FechaCreacion,
		UC.Nombre + ' ' + UC.Apellido AS UsuarioModificacion,
		D.FechaModificacion,
		1 AS CantidadDepositoUltimoCierre,
		ISNULL((SELECT
			SUM(T.TotalValidado) + SUM(T.TotalAValidar)
		FROM Dispositivo.Depositario AS DEP WITH(NOLOCK)
		INNER JOIN Operacion.Transaccion AS T WITH(NOLOCK) ON
			T.DepositarioId=DEP.Id
		INNER JOIN Operacion.Contenedor AS C WITH(NOLOCK) ON
			C.Id=T.ContenedorId
		WHERE 
			DEP.Id=D.Id
			AND C.FechaCierre >= GETDATE()
		),0) AS ValorTotalEnBolsa,
		CASE @SemaforoOnline
			WHEN CAST(1 AS BIT) THEN 'Verde'
			ELSE 'Rojo'
		END AS SemaforoOnline,
		CASE @SemaforoAnomalias
			WHEN CAST(1 AS BIT) THEN 'Verde'
			ELSE 'Rojo'
		END AS SemaforoAnomalia,
		CASE
			WHEN CxB.PorcentajeOcupacionBolsa < 70 THEN 'Verde'
			WHEN CxB.PorcentajeOcupacionBolsa BETWEEN 70 AND 90 THEN 'Amarillo'
			ELSE 'Rojo'
		END AS SemaforoOcupacionBolsa
	FROM Dispositivo.Depositario AS D WITH(NOLOCK)
	INNER JOIN Directorio.Sector AS SEC WITH(NOLOCK) ON
		SEC.Id=D.SectorId
	INNER JOIN Dispositivo.Modelo AS M WITH(NOLOCK) ON
		D.ModeloId=M.Id
	INNER JOIN Directorio.Sucursal AS SUC WITH(NOLOCK) ON
		SUC.Id=SEC.SucursalId
	INNER JOIN Directorio.Empresa AS E WITH(NOLOCK) ON
		E.Id=SUC.EmpresaId
	LEFT JOIN Seguridad.Usuario AS UC WITH(NOLOCK) ON
		UC.Id=D.UsuarioCreacion
	LEFT JOIN Seguridad.Usuario AS UM WITH(NOLOCK) ON
		UM.Id=D.UsuarioCreacion
	LEFT JOIN (SELECT
				   ISNULL(SUM(TD.CantidadUnidades) * 100 / (SELECT CAST(Capacidad AS FLOAT) FROM Operacion.TipoContenedor AS TC WITH(NOLOCK) WHERE TC.Id=C.TipoId),0) AS PorcentajeOcupacionBolsa,
				   D.Id AS DepositarioId
			   FROM Dispositivo.Depositario AS D WITH(NOLOCK)
			   LEFT JOIN Operacion.Transaccion AS T WITH(NOLOCK) ON
				   T.DepositarioId=D.Id
			   LEFT JOIN Operacion.TransaccionDetalle AS TD WITH(NOLOCK) ON
				   TD.TransaccionId=T.Id
			   LEFT JOIN Operacion.Contenedor AS C WITH(NOLOCK) ON
				   C.Id=T.ContenedorId
				   AND C.FechaCierre>=GETDATE()
			   WHERE
				   D.Id=D.Id
			   GROUP BY
				   D.Id,C.TipoId
			  ) AS CxB ON CxB.DepositarioId=D.Id


END

CREATE PROCEDURE [Operacion].[ObtenerDetalleTransaccion]
@TransaccionID BIGINT

--exec [Operacion].[ObtenerDetalleTransaccion] 5

AS
BEGIN

	SELECT
		TD.Id AS TransaccionDetalleId,
		TD.Fecha AS FechaTransaccionDetalle,
		D.Nombre AS Denominacion,
		D.Imagen AS ImagenDenominacion,
		M.Nombre AS Moneda,
		TD.CantidadUnidades,
		TD.TransaccionId
	FROM Operacion.TransaccionDetalle AS TD WITH(NOLOCK)
	INNER JOIN Valor.Denominacion AS D WITH(NOLOCK) ON
		TD.DenominacionId=D.Id
	INNER JOIN Valor.Moneda AS M WITH(NOLOCK) ON
		M.Id=D.MonedaId
	WHERE
		TD.TransaccionId=@TransaccionID
	ORDER BY
		TD.Fecha DESC
END

-- ======================================================================
-- Author:		Germ√°n Sivit
-- Create date: 16/5/2022
-- Description:	Devuelve todas las transacciones del depositario indicado
-- Uso: EXEC [Operacion].[ObtenerTransaccionesPorDepositario] 1
-- ======================================================================
CREATE PROCEDURE [Operacion].[ObtenerTransaccionesPorDepositario]
@DepositarioId BIGINT

--exec [Operacion].[ObtenerTransaccionesPorDepositario] 1

AS
BEGIN

	SELECT
		T.Id AS TransaccionId,
		T.Fecha AS FechaTransaccion,
		TT.Descripcion AS TipoTransaccion,
		UT.NickName AS UsuarioTransaccion,
		USC.Nombre + ' ' + USC.Apellido AS UsuarioCuenta,
		CTA.Nombre AS Cuenta,
		B.Descripcion AS Banco,
		C.Identificador AS Contenedor,
		T.TotalValidado,
		T.DepositarioId AS DepositarioId
	FROM Operacion.Transaccion AS T WITH(NOLOCK)
	INNER JOIN Operacion.TipoTransaccion AS TT WITH(NOLOCK) ON
		TT.Id=T.TipoId
	INNER JOIN Banca.UsuarioCuenta AS UC WITH(NOLOCK) ON
		UC.Id=T.UsuarioCuentaId
	INNER JOIN Banca.Cuenta AS CTA WITH(NOLOCK) ON
		CTA.Id=UC.CuentaId
	INNER JOIN Banca.Banco AS B WITH(NOLOCK) ON
		B.Id=CTA.BancoId
	INNER JOIN Operacion.Contenedor AS C WITH(NOLOCK) ON
		C.Id=T.ContenedorId
	INNER JOIN Operacion.TipoContenedor AS TC WITH(NOLOCK) ON
		TC.Id=C.TipoId
	INNER JOIN Operacion.CierreDiario AS CD WITH(NOLOCK) ON
		CD.Id=T.CierreDiarioId
	LEFT JOIN Seguridad.Usuario AS UT WITH(NOLOCK) ON
		UT.Id=T.UsuarioId
	LEFT JOIN Seguridad.Usuario AS USC WITH(NOLOCK) ON
		USC.Id=UC.UsuarioId
	WHERE
		T.DepositarioId=@DepositarioId
		AND T.TipoId <> 3 --3:Sobre
	ORDER BY
		T.Fecha DESC

END
